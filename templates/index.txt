    <script>
        const NUM_SENSORS = {{ num_sensors }};
        const MUSCLE_LABELS = {{ muscle_labels | tojson }};
        const LIVE_DATA_INTERVAL = 100; // ms
        // --- Increased X-axis duration to 5 seconds ---
        const MAX_POINTS_PER_CHART = 10000; // 5 seconds * 2000 Hz
        // -------------------------------
        const streamButton = document.getElementById('streamButton');
        const recordButton = document.getElementById('recordButton');
        const statusDiv = document.getElementById('status');
        const yAxisSlider = document.getElementById('yAxisSlider');
        const yAxisValue = document.getElementById('yAxisValue');
        let isRequestPending = false;
        let isCurrentlyStreaming = false; // Tracks streaming state
        let isCurrentlyRecording = false; // Tracks recording state
        let charts = [];
        let liveDataIntervalId = null;

        // --- Initialize Charts ---
        function initializeCharts() {
            const container = document.getElementById('chartsContainer');
            container.innerHTML = '';
            charts = [];
            for (let i = 0; i < NUM_SENSORS; i++) {
                const wrapper = document.createElement('div');
                wrapper.className = 'chart-wrapper';
                const titleDiv = document.createElement('div');
                titleDiv.className = 'chart-title';
                titleDiv.id = `chart-title-${i}`;
                titleDiv.textContent = MUSCLE_LABELS[i] || `Channel ${i}`;
                wrapper.appendChild(titleDiv);
                const chartDiv = document.createElement('div');
                chartDiv.className = 'chart-container';
                const canvas = document.createElement('canvas');
                canvas.id = `chart-${i}`;
                chartDiv.appendChild(canvas);
                wrapper.appendChild(chartDiv);
                container.appendChild(wrapper);
                const ctx = canvas.getContext('2d');
                const chart = new Chart(ctx, {
                    type: 'line',
                    data: {
                        labels: [],
                        datasets: [{
                            label: 'EMG Amplitude',
                            data: [],
                            borderColor: 'rgb(75, 192, 192)',
                            backgroundColor: 'rgba(75, 192, 192, 0.1)',
                            borderWidth: 1,
                            pointRadius: 0,
                            fill: true
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        animation: false,
                        scales: {
                            x: {
                                display: false,
                                type: 'linear',
                                min: 0,
                                // max will be set dynamically
                            },
                            y: {
                                beginAtZero: true,
                                suggestedMax: parseFloat(yAxisSlider.value),
                                suggestedMin: -parseFloat(yAxisSlider.value) // Show negative range too
                            }
                        },
                        plugins: {
                            legend: { display: false },
                            tooltip: { enabled: false }
                        }
                    }
                });
                charts.push(chart);
            }
        }

        // --- Update Charts with Live Data ---
        async function updateCharts() {
            // Update charts if streaming is active (regardless of recording)
            if (!isCurrentlyStreaming) return;
            try {
                const response = await fetch('/live_data');
                if (!response.ok) {
                     throw new Error(`HTTP error! status: ${response.status}`);
                }
                const data = await response.json();
                const liveDataChunks = data.data;
                const labels = data.labels;

                if (!liveDataChunks || liveDataChunks.length !== NUM_SENSORS) {
                    console.warn("Received unexpected live data format.");
                    return;
                }

                for (let ch = 0; ch < NUM_SENSORS; ch++) {
                    const chart = charts[ch];
                    const channelData = liveDataChunks[ch];
                    const label = labels[ch];
                    const titleElement = document.getElementById(`chart-title-${ch}`);

                    if (titleElement && titleElement.textContent !== label) {
                        titleElement.textContent = label;
                    }

                    if (channelData && channelData.length > 0) {
                        // Always show the last MAX_POINTS_PER_CHART points
                        let startIndex = Math.max(0, channelData.length - MAX_POINTS_PER_CHART);
                        let endIndex = channelData.length;
                        let displayData = channelData.slice(startIndex, endIndex);

                        // X-axis labels: 0 to displayData.length - 1
                        let xLabels = [];
                        for (let i = 0; i < displayData.length; i++) {
                            xLabels.push(i);
                        }

                        chart.data.labels = xLabels;
                        chart.data.datasets[0].data = displayData;
                        // Update x-axis max to fit the data
                        chart.options.scales.x.max = Math.max(1, displayData.length - 1);
                        chart.update(); // Efficient update
                    } else {
                         // Clear chart if no data
                         chart.data.labels = [];
                         chart.data.datasets[0].data = [];
                         chart.update();
                    }
                }
            } catch (error) {
                console.error("Error fetching or updating live charts:", error);
                // Optionally update status here if error is critical
            }
        }

        // --- Start/Stop Live Updates ---
        function startLiveUpdates() {
            if (liveDataIntervalId) {
                clearInterval(liveDataIntervalId);
            }
            // Start fetching live data periodically
            liveDataIntervalId = setInterval(updateCharts, LIVE_DATA_INTERVAL);
            console.log("Live chart updates started.");
        }

        function stopLiveUpdates() {
            if (liveDataIntervalId) {
                clearInterval(liveDataIntervalId);
                liveDataIntervalId = null;
                console.log("Live chart updates stopped.");
            }
            // Clear charts when stopping updates
            for (let i = 0; i < charts.length; i++) {
                 const chart = charts[i];
                 chart.data.labels = [];
                 chart.data.datasets[0].data = [];
                 chart.update();
            }
        }

        // --- Streaming Control ---
        streamButton.addEventListener('click', toggleStreaming);

        async function toggleStreaming() {
            if (isRequestPending) return;
            isRequestPending = true;

            const action = isCurrentlyStreaming ? 'stop' : 'start';
            const buttonText = action === 'start' ? "Starting..." : "Stopping...";
            streamButton.textContent = buttonText;
            streamButton.disabled = true;
            updateStatus(`${action.charAt(0).toUpperCase() + action.slice(1)}ing streaming...`, 'info');

            try {
                const response = await fetch('/toggle_streaming', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ action: action })
                });

                const data = await response.json();
                if (data.success) {
                    isCurrentlyStreaming = !isCurrentlyStreaming; // Toggle state
                    updateStreamButtonState(isCurrentlyStreaming);
                    recordButton.disabled = !isCurrentlyStreaming; // Enable record only if streaming

                    if (isCurrentlyStreaming) {
                        updateStatus("Streaming started. Press and hold 'Record' to capture data.", 'success');
                        startLiveUpdates(); // Start fetching live data
                    } else {
                        updateStatus("Streaming stopped.", 'info');
                        stopLiveUpdates(); // Stop fetching and clear charts
                        // Ensure recording is also stopped if streaming stops
                        if (isCurrentlyRecording) {
                             isCurrentlyRecording = false;
                             updateRecordButtonState(false, "Hold to Record");
                             updateStatus("Streaming stopped, recording cancelled.", 'warning');
                        }
                    }
                } else {
                    updateStatus('Error: ' + data.message, 'error');
                    updateStreamButtonState(isCurrentlyStreaming); // Revert button state
                    recordButton.disabled = !isCurrentlyStreaming;
                }
            } catch (error) {
                console.error('Streaming toggle error:', error);
                updateStatus('Error: Failed to connect to server.', 'error');
                updateStreamButtonState(isCurrentlyStreaming); // Revert button state
                recordButton.disabled = !isCurrentlyStreaming;
            } finally {
                isRequestPending = false;
                // Re-enable button after a short delay to prevent double clicks
                setTimeout(() => {
                     if (!isRequestPending) { // Check again in case another request started
                         streamButton.disabled = false;
                     }
                }, 500);
            }
        }

        function updateStreamButtonState(isStreaming) {
            streamButton.textContent = isStreaming ? "Stop Streaming" : "Start Streaming";
            if (isStreaming) {
                streamButton.classList.add('streaming');
            } else {
                streamButton.classList.remove('streaming');
            }
        }

        // --- Recording Control (Segment) ---
        recordButton.addEventListener('mousedown', startRecordingSegment);
        recordButton.addEventListener('touchstart', function (e) {
            e.preventDefault();
            startRecordingSegment();
        });
        window.addEventListener('mouseup', stopRecordingSegment);
        window.addEventListener('touchend', stopRecordingSegment);

        async function startRecordingSegment() {
            // Prevent starting recording if not streaming or request pending
            if (isRequestPending || !isCurrentlyStreaming || isCurrentlyRecording) return;
            isRequestPending = true;

            updateRecordButtonState(true, "Recording...");
            updateStatus("Starting recording segment...", 'info');

            try {
                const response = await fetch('/start_recording', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' }
                });
                const data = await response.json();
                if (data.success) {
                    isCurrentlyRecording = true;
                    updateStatus(data.message, 'success'); // e.g., "Recording segment started (Trial #1)."
                     // Live updates should already be running from streaming start
                } else {
                    updateStatus('Error: ' + data.message, 'error');
                    updateRecordButtonState(false, "Hold to Record");
                }
            } catch (error) {
                console.error('Start recording segment error:', error);
                updateStatus('Error: Failed to connect to server.', 'error');
                updateRecordButtonState(false, "Hold to Record");
            } finally {
                isRequestPending = false;
            }
        }

        async function stopRecordingSegment() {
            if (!isCurrentlyRecording || isRequestPending) return;
            isRequestPending = true;

            updateStatus("Stopping recording segment...", 'info');

            try {
                const response = await fetch('/stop_recording', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' }
                });
                const data = await response.json();
                if (data.success) {
                    updateStatus(data.message, 'success'); // e.g., "Recording segment #1 saved successfully."
                } else {
                    updateStatus('Warning: ' + data.message, 'error');
                }
            } catch (error) {
                console.error('Stop recording segment error:', error);
                updateStatus('Error: Failed to connect to server.', 'error');
            } finally {
                isCurrentlyRecording = false;
                updateRecordButtonState(false, "Hold to Record");
                isRequestPending = false;
            }
        }

        function updateRecordButtonState(isRecording, text) {
            recordButton.textContent = text;
            if (isRecording) {
                recordButton.classList.add('recording');
            } else {
                recordButton.classList.remove('recording');
            }
        }

        function updateStatus(message, typeClass = '') {
            statusDiv.textContent = message;
            statusDiv.className = typeClass; // Apply CSS class for color
        }

        // --- Y-Axis Slider Control ---
        function updateYAxisMax() {
            const newMax = parseFloat(yAxisSlider.value);
            // Display with 4 decimal places for smaller values
            yAxisValue.textContent = newMax.toFixed(4);
            for (let i = 0; i < charts.length; i++) {
                const chart = charts[i];
                if (chart && chart.options?.scales?.y) {
                    chart.options.scales.y.suggestedMax = newMax;
                    chart.options.scales.y.suggestedMin = -newMax; // Show negative range
                    chart.update();
                }
            }
        }
        yAxisSlider.addEventListener('input', updateYAxisMax);

        // --- Initialize on Load ---
        window.onload = function () {
            initializeCharts();
            updateStatus("System ready. Click 'Start Streaming' to begin visualizing data.", 'info');
            // Initialize with 4 decimals for the initial value
            yAxisValue.textContent = parseFloat(yAxisSlider.value).toFixed(4);
            // Disable record button initially
            recordButton.disabled = true;
        };
    </script>
